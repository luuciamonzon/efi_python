
#Clonar el proyecto:
    git clone https://https://github.com/luuciamonzon/efi_python.git
    cd efi_python

    python3 -m venv .env

    source .env/bin/activate
#Instalar requerimientos:
    pip install -r requirements.txt
#Correr el proyecto:
    flask run --reload
